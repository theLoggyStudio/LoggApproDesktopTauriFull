  dbapyellow.db …      — bases « couleur » ; anciennement dbla*.db

[Frontend]

  PageConnection -> SessionContext (sessionStorage)
  App (HashRouter)
    /connection -> PageConnection
    /stock (ProtectedRoute)
      StockLayout (Sider + Outlet)

        /stock/movements        : stock_list_movements, stock_add_movement
      StockDbSettingsModal     : stock_test_remote_db
  AntdThemeBridge : ConfigProvider + theme <- themes[themeNumber]
  ThemeCssSync     : CSS variables --primary, --secondary, ... sur <html>
